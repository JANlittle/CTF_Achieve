代码文件：CommonController.class.php、SystemController.class.php

//获取配置
$system_config = S('ALL_CONFIG'.C('SITEID'));
//空的情况下生成缓存
if(empty($system_config)){
    //生成缓存
    setConfig('ALL_CONFIG'.C('SITEID'),array(0,C('SITEID')));
    $system_config = S('ALL_CONFIG'.C('SITEID'));
}
//缓存设置成配置
C($system_config);
$this->system_config = $system_config;
